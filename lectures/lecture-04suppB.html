You can change the shape of points in ggplot:


df <- expand.grid(x = 1:3, y = 1:2) %>% 
  mutate(id = factor(1:n()))

ggplot(df, aes(x, y, shape = id)) + 
  geom_point(size = 4) + 
  theme_void() +
  guides(shape = FALSE) + 
  theme(plot.background = element_rect(fill = "grey")) +
  xlim(0.5, 3.5) + ylim(0.5, 2.5)
